** view valid date from pem file
   #+BEGIN_SRC sh
     % openssl x509 -startdate -enddate -noout -in apns.pem
     notBefore=Sep 20 01:01:06 2016 GMT
     notAfter=Oct 20 01:01:06 2017 GMT
   #+END_SRC
