* git memo
** fetch all and prune
   #+BEGIN_SRC sh
     git fetch --all --tags --prune
   #+END_SRC
** remove merged branches
   #+BEGIN_SRC sh
     git branch --merged=develop|grep -v '*'|grep -v develop|grep -v master|xargs git branch -d
   #+END_SRC
** save credential
   #+BEGIN_SRC sh
     git config credential.helper store
   #+END_SRC
